Команды для работы в Linux терминале

Файлы  

Ls [ключ] [файл]
ls	Список файлов в каталоге
ls -l	Список файлов в каталоге в табличном виде
ls –a 	Список файлов в каталоге, включая скрытые папки
ls   -author	Список файлов + имя создателя
ls -cl	Список файлов с сортировкой по имени
ls -ct	Список файлов с сортировкой по времени создания (сначала самые но-вые)
ls -s	Список файлов с размером файла
ls -S	Список файлов с сортировкой по размерам (сначала большие)
ls *	Список текстовых файлов
ls –l >file.txt	Запись списка файлов в file.txt
	
pwd	Показать, где находишься
	
cd file.txt	Перемещение в файл file.txt
cd ..	Перемещение на папку вверх
cd -	Перемещение в предыдущую папку, в которой были (вверх или вниз)
cd /	Перемещение в корневую папку
	
cat [ключ] [файл]
cat file	Файл file выводится в терминал
cat file1 file2 > file3
	Сцепление (слияние) файлов file1 и file2, после чего они помещаются в файл file3
cat file1 >> file2
	Присоединение файла file1 к концу файла file2
cat -b file1
	Вывод содержимого файла с нумерацией строк, но без нумерации пу-стых строк
cat -n file1
	Нумерация всех строк
	
mkdir папка	Создать каталог (папку)
mkdir папка1 пап-ка2 	Создать 2 папки
mkdir {пап-ка1,папка2}
	Создать 2 папки (запись без пробелов)
rmdir папка1	Удалить папку1
rm –d папка1	Удалить папку1
	
Touch file1	Создать файл1
Touch file1 file2	Создать 2 файлы
>file1	Создать файл1 
>папка/файл	Создать файл в папке (путь)
rm file1	Удалить файл1
	
copy file1 file2	Копировать file1 с новым именем file2
	
more file1	выводит текста из файла на экран по 24 строки, нажатие  [Пробел]  выво-дит очередную порции. Досрочно завершает работу при нажатии на клавишу [Q]
less file1	выводит текста из файла на экран, поддерживает возможности прокрут-ки вверх и поиска. Завершает работу при нажатии на клавишу [Q]
	
find file1	Поиск в данном каталоге
find file1 (что) пап-ка(где)	Поиск по заданному пути
find / -name f|ile1	Поиск по всем каталогам (с sudo) 
	
mv file1 file2	Переименовать file1 в file2
	
sudo !!	Повтор предыдущей команды
